The system allows teachers to input student data, calculate statistical measures (average, median, min, max), assign letter grades based on predefined criteria, filter pass/fail students, and display results in a formatted table. ## Complete Python Code

```python
import statistics

# Student Information Header
STUDENT_NAME = "Ayush Kumar"
ROLL_NUMBER = "2501730366"
PASS_MARK = 40

def welcome_message():
    """Display welcome message and project title"""
    print("\n" + "="*70)
    print("GRADEBOOK ANALYZER - Student Performance Analysis Tool".center(70))
    print("="*70)
    print(f"Submitted by: {STUDENT_NAME}")
    print(f"Roll Number: {ROLL_NUMBER}")
    print("="*70 + "\n")

def get_student_data():
    """Task 2: Get student names and marks from user input"""
    marks = {}
    try:
        num_students = int(input("Enter the number of students in the class: "))
        if num_students <= 0:
            print("Error: Number of students must be greater than 0")
            return get_student_data()
        
        for i in range(num_students):
            name = input(f"Enter student {i+1} name: ").strip()
            if not name:
                print("Error: Name cannot be empty. Please try again.")
                i -= 1
                continue
            
            try:
                mark = float(input(f"Enter {name}'s marks (0-100): "))
                if mark < 0 or mark > 100:
                    print("Error: Marks must be between 0 and 100. Please try again.")
                    i -= 1
                    continue
                marks[name] = mark
            except ValueError:
                print("Error: Please enter a valid number for marks.")
                i -= 1
                continue
        
        return marks
    except ValueError:
        print("Error: Please enter a valid number for student count.")
        return get_student_data()

def calculate_average(marks):
    """Task 3: Calculate average marks"""
    if not marks:
        return 0
    return sum(marks.values()) / len(marks)

def calculate_median(marks):
    """Task 3: Calculate median marks"""
    if not marks:
        return 0
    marks_list = list(marks.values())
    return statistics.median(marks_list)

def find_max_score(marks):
    """Task 3: Find maximum score"""
    if not marks:
        return 0
    return max(marks.values())

def find_min_score(marks):
    """Task 3: Find minimum score"""
    if not marks:
        return 0
    return min(marks.values())

def display_statistics(marks):
    """Task 3: Display all statistical analysis"""
    if not marks:
        print("No student data available.")
        return
    
    avg = calculate_average(marks)
    med = calculate_median(marks)
    max_score = find_max_score(marks)
    min_score = find_min_score(marks)
    
    print("\n" + "-"*50)
    print("STATISTICAL ANALYSIS".center(50))
    print("-"*50)
    print(f"Average Marks: {avg:.2f}")
    print(f"Median Marks: {med:.2f}")
    print(f"Maximum Score: {max_score:.2f}")
    print(f"Minimum Score: {min_score:.2f}")
    print(f"Total Students: {len(marks)}")
    print("-"*50 + "\n")

def assign_grade(mark):
    """Task 4: Assign grade based on marks"""
    if mark >= 90:
        return "A"
    elif mark >= 80:
        return "B"
    elif mark >= 70:
        return "C"
    elif mark >= 60:
        return "D"
    else:
        return "F"

def get_grades(marks):
    """Task 4: Create grades dictionary and count distribution"""
    grades = {name: assign_grade(mark) for name, mark in marks.items()}
    
    # Count students per grade
    grade_count = {"A": 0, "B": 0, "C": 0, "D": 0, "F": 0}
    for grade in grades.values():
        grade_count[grade] += 1
    
    return grades, grade_count

def display_grade_distribution(grade_count):
    """Task 4: Display grade distribution"""
    print("\n" + "-"*50)
    print("GRADE DISTRIBUTION".center(50))
    print("-"*50)
    for grade in ["A", "B", "C", "D", "F"]:
        print(f"Grade {grade}: {grade_count[grade]} students")
    print("-"*50 + "\n")

def get_pass_fail_students(marks):
    """Task 5: Use list comprehensions to filter pass/fail students"""
    passed_students = [name for name, mark in marks.items() if mark >= PASS_MARK]
    failed_students = [name for name, mark in marks.items() if mark < PASS_MARK]
    
    return passed_students, failed_students

def display_pass_fail(passed_students, failed_students):
    """Task 5: Display pass/fail students"""
    print("\n" + "-"*50)
    print("PASS/FAIL ANALYSIS".center(50))
    print("-"*50)
    print(f"Passed Students ({len(passed_students)}): {', '.join(passed_students) if passed_students else 'None'}")
    print(f"Failed Students ({len(failed_students)}): {', '.join(failed_students) if failed_students else 'None'}")
    print("-"*50 + "\n")

def display_results_table(marks, grades):
    """Task 6: Display formatted results table"""
    print("\n" + "="*70)
    print("COMPLETE RESULTS TABLE".center(70))
    print("="*70)
    print(f"{'Name':<20} {'Marks':<15} {'Grade':<15}")
    print("-"*70)
    
    for name in sorted(marks.keys()):
        mark = marks[name]
        grade = grades[name]
        print(f"{name:<20} {mark:<15.2f} {grade:<15}")
    
    print("="*70 + "\n")

def run_analysis():
    """Task 6: Main function to run complete analysis"""
    welcome_message()
    
    # Get student data
    marks = get_student_data()
    
    if not marks:
        print("No student data entered. Exiting...")
        return
    
    # Task 3: Display statistics
    display_statistics(marks)
    
    # Task 4: Assign grades and display distribution
    grades, grade_count = get_grades(marks)
    display_grade_distribution(grade_count)
    
    # Task 5: Pass/Fail analysis
    passed_students, failed_students = get_pass_fail_students(marks)
    display_pass_fail(passed_students, failed_students)
    
    # Task 6: Display results table
    display_results_table(marks, grades)

def main_menu():
    """Task 6: Main menu with loop for re-running analysis"""
    while True:
        run_analysis()
        
        choice = input("Do you want to analyze another class? (yes/no): ").strip().lower()
        if choice != "yes" and choice != "y":
            print("\nThank you for using GradeBook Analyzer. Goodbye!")
            print("="*70 + "\n")
            break

if __name__ == "__main__":
    main_menu()
```
